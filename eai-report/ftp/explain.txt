            <xsl:attribute name="ConnectionName">
                <xsl:value-of select="$Start/parameters/connectionName"/>
            </xsl:attribute>
            <xsl:if test="$Start/parameters/remoteLocation">
                <xsl:attribute name="requestURI">
                    <xsl:value-of select="$Start/parameters/remoteLocation"/>
                </xsl:attribute>
            </xsl:if>
            <ns7:nbMaxRetry>
                <xsl:value-of select="5"/>
            </ns7:nbMaxRetry>
            <ns7:waitForRetry>
                <xsl:value-of select="30"/>
            </ns7:waitForRetry>
            <ns7:batchSchedulerInfo>
                <xsl:value-of select="false()"/>
            </ns7:batchSchedulerInfo>
        </ns7:Header>
        <ns7:File>
            <xsl:attribute name="Location">
                <xsl:value-of select="$Start/parameters/fileLocation"/>
            </xsl:attribute>
            <xsl:attribute name="Name">
                <xsl:value-of select="$Start/parameters/fileName"/>
            </xsl:attribute>
            <xsl:if test="$Start/parameters/remoteLocation">
                <xsl:attribute name="RemoteLocation">
                    <xsl:value-of select="$Start/parameters/remoteLocation"/>
