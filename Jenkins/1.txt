// --- CONFIGURATION ---
def fixedInputs = [
  "bw-eai-bofi", 
 "bw-eai-bofi-swift",

 "bw-eai-bofi-batch",

// --- HELPERS ---
def esc = { s ->
  if (s == null) return ""
  s.toString().replace("&","&amp;").replace("<","&lt;").replace(">","&gt;").replace("\"","&quot;").replace("'","&#39;")
}

def fetchOptions = { String artifactOrKey ->
  try {
    def cmd = "curl -s -k https://src1ems02:9091/rest/livraison/getVersionList?art=${artifactOrKey}"
    def out = cmd.execute().text?.trim()
    def tokens = out ?
  out.tokenize(",")
     .collect { it.trim() }
     .findAll { !it.toUpperCase().contains("SNAPSHOT") }
  : []
    return tokens.isEmpty() ? ["(aucune option)"] : tokens
  } catch (e) {
    return ["Erreur API"]
  }
}

// --- GÉNÉRATION HTML ---

sb << """

.ac-wrap { max-width: 980px; font-family: -apple-system, system-ui, "Segoe UI", Roboto, sans-serif; }

.ac-input, .ac-select { width: 100%; padding: 6px; border: 1px solid rgba(0,0,0,.2); border-radius: 4px; }

        </select>

        <input type="checkbox" class="js-include-checkbox" />

    </tr>
  """

sb << """
      </tbody>
    </table>
  </div>
</div>

<script>
(function() {
    // 1. Fonction de mise à jour centrale
    function forceUpdate() {
        const container = document.getElementById('artifact-container');
        const finalInput = document.getElementById('final-json-output');
        
        if (!container || !finalInput) return;

        const rows = container.querySelectorAll('.artifact-row');
        const selection = [];

        rows.forEach(row => {
            const isChecked = row.querySelector('.js-include-checkbox').checked;
            if (isChecked) {
                selection.push({
                    artifact: row.getAttribute('data-name'),
                    version: row.querySelector('.js-version-select').value
                });
            }
        });

        const newValue = JSON.stringify(selection);
        if (finalInput.value !== newValue) {
            finalInput.value = newValue;
            // Notification pour Jenkins
            finalInput.dispatchEvent(new Event('change', { bubbles: true }));
        }
    }

    // 2. DÉLÉGATION D'ÉVÉNEMENTS : On écoute sur le document entier
    // Cela survit même si Jenkins vide et remplace le contenu du paramètre
    document.addEventListener('change', function(e) {
        // On ne réagit que si l'élément modifié est à l'intérieur de notre conteneur
        if (e.target && e.target.closest('#artifact-container')) {
            // Si c'est notre propre input caché qui change, on ignore pour éviter la boucle
            if (e.target.id === 'final-json-output') return;
            
            forceUpdate();
        }
    }, true); // 'true' pour capturer l'événement en phase de capture

    // 3. INITIALISATION ET SURVIE
    // On lance une mise à jour initiale
    forceUpdate();
    
    // On relance une mise à jour régulière (toutes les 2 sec) juste au cas où 
    // Jenkins aurait fait une mise à jour silencieuse du DOM sans déclencher 'change'
    setInterval(forceUpdate, 2000);
})();
</script>
"""

return sb.toString()
