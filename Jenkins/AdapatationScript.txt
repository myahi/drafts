stage("Restart") {
  steps {
    sh """
      set -euo pipefail
      ssh -o StrictHostKeyChecking=no ${DEPLOY_USR}@${TARGET_HOST} '
        set -e
        cd "'"${BIN_DIR}"'"

        if [ -x "./restart.sh" ]; then
          ./restart.sh "'"${ENV}"'" "'"${COULEUR}"'"
        else
          [ -x "./stop.sh" ] && ./stop.sh || true
          [ -x "./start.sh" ] && ./start.sh "'"${ENV}"'" "'"${COULEUR}"'"
        fi
      '
    """
  }
}
