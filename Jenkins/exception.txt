import groovy.json.JsonSlurper
import java.util.Base64
import jenkins.model.Jenkins
import com.cloudbees.plugins.credentials.CredentialsProvider
import com.cloudbees.plugins.credentials.common.StandardUsernamePasswordCredentials

def appBase   = (binding.variables.get('APP') ?: '').toString().trim()
def artifact  = "${appBase}-conf"                 // IMPORTANT : -conf
def groupPath = "fr/labanquepostale/marches/eai"

// repo keys uniquement
def repos = [
  "eai-marches-0002775-maven-local-sas"
]

def baseUrl = (binding.variables.get('EAI_ARTIFACTORY_URL') ?: '').toString().trim()
def CRED_ID = "eai-marches-0002775-tech-ci-snapshot-user"

if (!artifact || !baseUrl) return ["(missing params)"]

if (baseUrl.endsWith("/")) baseUrl = baseUrl[0..-2]
if (!baseUrl.contains("/artifactory")) baseUrl = baseUrl + "/artifactory"

// credentials
def creds = CredentialsProvider.lookupCredentials(
  StandardUsernamePasswordCredentials,
  Jenkins.instance,
  null,
  null
).find { it.id == CRED_ID }

if (!creds) return ["(credential not found: ${CRED_ID})"]

def user  = creds.username
def token = creds.password?.plainText
if (!user || !token) return ["(empty credential)"]

def auth = Base64.encoder.encodeToString("${user}:${token}".getBytes("UTF-8"))

def fetchRepo = { String repo ->
  def aql = """
items.find({
  "repo": "${repo}",
  "path": {"\\\$match":"${groupPath}/${artifact}/*"},
  "name": {"\\\$match":"${artifact}-*.zip"}
}).include("repo","path","name")
"""
  def url = new URL("${baseUrl}/api/search/aql")
  def conn = (HttpURLConnection) url.openConnection()
  conn.setRequestMethod("POST")
  conn.setDoOutput(true)
  conn.setConnectTimeout(5000)
  conn.setReadTimeout(15000)
  conn.setRequestProperty("Authorization", "Basic ${auth}")
  conn.setRequestProperty("Content-Type", "text/plain; charset=UTF-8")
  conn.outputStream.withWriter("UTF-8") { it << aql }

  int rc = conn.responseCode
  def stream = (rc >= 200 && rc < 300) ? conn.inputStream : conn.errorStream
  def payload = stream ? stream.getText("UTF-8") : ""

  if (!(rc >= 200 && rc < 300)) {
    // IMPORTANT: renvoyer une String "pure"
    return [("(repo=${repo}) HTTP ${rc} ${payload?.take(200) ?: ''}".toString())]
  }

  def json = new JsonSlurper().parseText(payload)
  return (json?.results ?: [])
}

def results = repos.collectMany { fetchRepo(it) }

// 1) extraire les erreurs (String/GString/CharSequence) sans casser
def errors = results.findAll { !(it instanceof Map) }
                   .collect { it.toString() }

// 2) ne garder que les vrais résultats AQL (Map avec path)
def items = results.findAll { it instanceof Map && it.path != null && it.name != null }

if (!items && errors) return errors

// 3) extraire les versions depuis le path
def prefix = "${groupPath}/${artifact}/"
def versions = items.collect { r ->
  def p = r.path as String
  if (p.startsWith(prefix)) p.substring(prefix.length()).tokenize('/')[0] else null
}.findAll { it }.unique()

def out = versions.collect { it.toString() }.sort().reverse()

// Si erreurs + versions, tu peux choisir de les montrer ou pas.
// Là je n'affiche les erreurs que s'il n'y a aucune version.
return out ?: (errors ?: ["(no versions found)"])
