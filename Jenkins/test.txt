
stage('Modules resolution') {
  steps {
    script {
      if (!fileExists('pom.xml')) {
        error("pom.xml introuvable, projet non Maven")
      }

      def parentPomText = readFile('pom.xml')
      def moduleDirs = extractModuleDirs(parentPomText)

      if (!moduleDirs || moduleDirs.isEmpty()) {
        // mono-module STRICT
        env.MODULES = params.APP
        env.IS_MULTI = "false"
        echo "Mono-module detected"
        return
      }

      def artifactIds = []
      moduleDirs.each { m ->
        def p = "${m}/pom.xml"
        if (!fileExists(p)) {
          error("pom.xml introuvable pour le module: ${m}")
        }
        artifactIds << artifactIdFromPom(readFile(p))
      }

      artifactIds = artifactIds.unique()

      if (artifactIds.isEmpty()) {
        error("Modules déclarés mais aucun artifactId détecté")
      }

      // ✅ MULTI = UNIQUEMENT les modules
      env.MODULES = artifactIds.join(',')
      env.IS_MULTI = "true"

      echo "Modules multi détectés = ${env.MODULES}"
    }
  }
}



def parentPackaging = readPackagingFromPom(readFile('pom.xml'))

@NonCPS
def readPackagingFromPom(String pomText) {
  def pom = new XmlSlurper(false, false).parseText(pomText)
  def p = pom.depthFirst()
             .find { it.name() == 'packaging' }
             ?.text()
             ?.trim()
  return p ?: 'jar'
}

@NonCPS
def extractModuleDirs(String pomText) {
  def pom = new XmlSlurper(false, false).parseText(pomText)
  pom.depthFirst()
     .findAll { it.name() == 'module' }
     .collect { it.text().trim() }
     .findAll { it }
}

@NonCPS
def artifactIdFromPom(String pomText) {
  def pom = new XmlSlurper(false, false).parseText(pomText)
  def aid = pom.depthFirst()
               .find { it.name() == 'artifactId' }
               ?.text()
               ?.trim()
  if (!aid) throw new RuntimeException("artifactId introuvable dans pom")
  return aid
}

pipeline {
  agent any
  options { timestamps(); disableConcurrentBuilds() }

  environment {
    ART_CRED = 'eai-marches-0002775-tech-ci-snapshot-user'
    GROUP_PATH = 'fr/labanquepostale/marches/eai'
    JAR_REPO  = 'eai-marches-0002775-maven-local-sas'
    CONF_REPO = 'eai-marches-0002775-maven-local-sas'
    GIT_CRED = 'gitlab_read_repo_token'   // (tu utilises usr_gitlab_eai dans checkout: je garde TON checkout tel quel plus bas)
    SSH_OPTS = '-q -o BatchMode=yes -o StrictHostKeyChecking=no -o ConnectTimeout=5'
  }

  stages {

    stage('Resolve & validate inputs') {
      steps {
        script {
          def MATRIX = [
            "eai-camel-rgv": [
              dev:     [ mono: ['sdv4eai02'] ],
              recette: [ mono: ['src4eai05'] ],
              preprod: [ mono: ['spp4eai02'] ],
              prod:    [ mono: ['spr4eai02'] ]
            ],
            "eai-camel-report": [
              dev:     [ mono: ['sdv4eai02'] ],
              recette: [ mono: ['src4eai05'] ],
              preprod: [ mono: ['spp4eai02'] ],
              prod:    [ mono: ['spr4eai02'] ]
            ]
          ]

          def needsJar  = (params.DEPLOY_MODE in ['BIN_AND_CONF','BIN_ONLY'])
          def needsConf = (params.DEPLOY_MODE in ['BIN_AND_CONF','CONF_ONLY'])

          if (needsJar  && !params.JAR_VERSION?.trim())  error("JAR_VERSION requis pour ${params.DEPLOY_MODE}")
          if (needsConf && !params.CONF_VERSION?.trim()) error("CONF_VERSION requis pour ${params.DEPLOY_MODE}")

          env.NEEDS_JAR  = needsJar.toString()
          env.NEEDS_CONF = needsConf.toString()

          def envMap = MATRIX[params.APP]?.get(params.ENV)
          if (!envMap) error("ENV non supporté: APP=${params.APP} ENV=${params.ENV}")

          def allowed = envMap.keySet().toList().sort()
          if (!allowed.contains(params.INSTANCE)) {
            error("INSTANCE '${params.INSTANCE}' non autorisée pour ${params.APP}/${params.ENV}. Autorisées: ${allowed}")
          }

          def targets = envMap[params.INSTANCE]
          if (!targets || targets.isEmpty()) error("Aucune VM configurée pour APP=${params.APP} ENV=${params.ENV} INSTANCE=${params.INSTANCE}")

          env.TARGET_VMS = targets.join(',')

          echo """
========== PLAN ==========
APP          : ${params.APP}
ENV          : ${params.ENV}
INSTANCE     : ${params.INSTANCE}
MODE         : ${params.DEPLOY_MODE}
TARGETS      : ${env.TARGET_VMS}
JAR_VERSION  : ${params.JAR_VERSION}
CONF_VERSION : ${params.CONF_VERSION}
========================
"""
        }
      }
    }

    stage('Checkout Git configuration') {
      steps {
        script {
          def branch = (params.ENV == 'prod') ? 'master' : 'develop'
          def repoUrl = "${env.EAI_REPO_URL}/${params.APP}.git"
          echo "Checkout repo=${repoUrl} branch=${branch}"

          checkout([
            $class: 'GitSCM',
            branches: [[name: branch]],
            userRemoteConfigs: [[
              url: repoUrl,
              credentialsId: 'usr_gitlab_eai'
            ]]
          ])
        }
      }
    }

    stage('Debug workspace') {
      steps {
        sh '''
          pwd
          ls -la
          echo "--- pom.xml files ---"
          find . -maxdepth 3 -name pom.xml -print
        '''
      }
    }

    stage('Modules resolution') {
      steps {
        script {
          if (!fileExists('pom.xml')) {
            // fallback mono
            env.MODULES = params.APP
            env.IS_MULTI = "false"
            echo "No pom.xml -> fallback mono: MODULES=${env.MODULES}"
            return
          }

          def parentPomText = readFile('pom.xml')
          def moduleDirs = extractModuleDirs(parentPomText)

          if (!moduleDirs || moduleDirs.isEmpty()) {
            // mono (ou parent sans modules)
            env.MODULES = params.APP
            env.IS_MULTI = "false"
            echo "Mono-module detected -> MODULES=${env.MODULES}"
            return
          }

          def artifactIds = []
          moduleDirs.each { m ->
            def p = "${m}/pom.xml"
            if (!fileExists(p)) error("pom.xml introuvable pour le module: ${m} (attendu: ${p})")
            artifactIds << artifactIdFromPom(readFile(p))
          }

          env.MODULES = artifactIds.unique().join(',')
          env.IS_MULTI = "true"

          echo "DEBUG moduleDirs=${moduleDirs}"
          echo "DEBUG artifactIds=${artifactIds.unique()}"
          echo "Multi-module detected -> MODULES=${env.MODULES}"
        }
      }
    }

    stage('Validate SSH connection') {
      steps {
        script {
          def hosts = env.TARGET_VMS.split(',')
          hosts.each { host ->
            sh """
              set -euo pipefail
              ssh ${env.SSH_OPTS} ${env.EAI_SSH_USER}@"${host}" "echo OK: \$(hostname) as \$(whoami)"
            """
          }
        }
      }
    }

    stage('Prepare remote deploy script') {
      steps {
        script {
          def scriptText = '''#!/usr/bin/env bash
set -euo pipefail

PARENT_APP="$1"
MODULE="$2"
ENV_NAME="$3"
INSTANCE="$4"
MODE="$5"
JAR_VERSION_UI="${6:-}"
CONF_VERSION_UI="${7:-}"
IS_MULTI="${8:-false}"

APPS_BASE="/serveur_apps"

APP_DIR="${APPS_BASE}/${PARENT_APP}"
INST_DIR="${APP_DIR}/${INSTANCE}"

if [[ "$IS_MULTI" == "true" ]]; then
  MOD_DIR="${INST_DIR}/${MODULE}"
else
  MOD_DIR="${INST_DIR}"
fi

BIN_DIR="${MOD_DIR}/bin"
LIB_DIR="${MOD_DIR}/lib"
CONFIG_DIR="${MOD_DIR}/config"
RUN_DIR="${MOD_DIR}/run"

APP_INSTANCE="${PARENT_APP}-${INSTANCE}"

COLOR=""
if [[ "$INSTANCE" != "mono" ]]; then
  COLOR="$INSTANCE"
fi

OLD_BIN="${BIN_DIR}"
stop_existing() {
  local s="stop.sh"
  if [[ -x "${OLD_BIN}/${s}" ]]; then
    echo "[REMOTE] Will run stop: ${OLD_BIN}/${s} '${APP_INSTANCE}'"
    "${OLD_BIN}/${s}" "${APP_INSTANCE}" || true
  else
    echo "[REMOTE] No stop.sh at ${OLD_BIN}/${s} (skip)"
  fi
}

echo "[REMOTE] Stopping (best effort) before cleanup..."
stop_existing

mkdir -p "$BIN_DIR" "$LIB_DIR" "$CONFIG_DIR" "$RUN_DIR"

run_script() {
  local s="$1"
  if [[ -x "$BIN_DIR/$s" ]]; then
    echo "[REMOTE] Run: $BIN_DIR/$s '${ENV_NAME}' '${APP_INSTANCE}'"
    "$BIN_DIR/$s" "$ENV_NAME" "$APP_INSTANCE"
  else
    echo "[REMOTE] Script not executable: $BIN_DIR/$s"
    return 127
  fi
}

# ----- CONF install -----
if [[ "$MODE" == "BIN_AND_CONF" || "$MODE" == "CONF_ONLY" ]]; then
  [[ -f /tmp/conf.zip ]] || { echo "ERROR: /tmp/conf.zip absent"; exit 2; }

  STAGE_DIR="$(mktemp -d "/tmp/${MODULE}-config-stage-XXXXXX")"
  unzip -o "/tmp/conf.zip" -d "$STAGE_DIR" >/dev/null

  # Zip structure: config/base/* and config/<module>/*
  COMMON_DIR="$STAGE_DIR/config/base"
  MOD_CONF_DIR="$STAGE_DIR/config/$MODULE"

  rm -rf "$CONFIG_DIR"/*
  mkdir -p "$CONFIG_DIR"

  if [[ -d "$COMMON_DIR" ]]; then
    cp -f "$COMMON_DIR/"* "$CONFIG_DIR/" 2>/dev/null || true
  fi

  if [[ -d "$MOD_CONF_DIR" ]]; then
    cp -f "$MOD_CONF_DIR/"* "$CONFIG_DIR/" 2>/dev/null || true
  else
    echo "WARN: No module-specific conf for '$MODULE' (expected: $MOD_CONF_DIR)"
  fi

  rm -rf "$STAGE_DIR"
fi

# ----- JAR install -----
if [[ "$MODE" == "BIN_AND_CONF" || "$MODE" == "BIN_ONLY" ]]; then
  [[ -f /tmp/app.jar ]] || { echo "ERROR: /tmp/app.jar absent"; exit 3; }
  mv -f "/tmp/app.jar" "$LIB_DIR/${MODULE}-${JAR_VERSION_UI}.jar"
  ln -sfn "$LIB_DIR/${MODULE}-${JAR_VERSION_UI}.jar" "$LIB_DIR/app.jar"
fi

# ----- BIN scripts: sync from ops repo on target VM -----
OPS_ROOT="/serveur_apps/eai-camel-ops"
OPS_BIN="${OPS_ROOT}/${PARENT_APP}/bin"
OPS_CONFIG="${OPS_ROOT}/${PARENT_APP}/config"

if [[ -d "$OPS_BIN" ]]; then
  cp -f "$OPS_BIN/"*.sh "$BIN_DIR/" 2>/dev/null || true
  chmod +x "$BIN_DIR/"*.sh 2>/dev/null || true
else
  echo "WARN: OPS bin missing: $OPS_BIN"
fi

if [[ -d "$OPS_CONFIG" ]]; then
  cp -f "$OPS_CONFIG/"* "$CONFIG_DIR/" 2>/dev/null || true
else
  echo "WARN: OPS config missing: $OPS_CONFIG"
fi

echo "[REMOTE] Scripts:"
ls -la "$BIN_DIR" || true

DEPLOYED_AT="$(date --iso-8601=seconds)"

cat > "$MOD_DIR/deployed.txt" <<EOF
parent=$PARENT_APP
module=$MODULE
instance=${PARENT_APP}-${INSTANCE}
env=$ENV_NAME
color=$COLOR
deployMode=$MODE
jarVersion=$JAR_VERSION_UI
confVersion=$CONF_VERSION_UI
deployedAt=$DEPLOYED_AT
isMulti=$IS_MULTI
EOF

if [[ -x "$BIN_DIR/restart.sh" ]]; then
  run_script restart.sh
else
  run_script start.sh
fi
'''
          writeFile file: 'deploy_remote.sh', text: scriptText
          sh 'chmod +x deploy_remote.sh'
        }
      }
    }

    stage('Deploy') {
      steps {
        script {
          // --- build module list
          def rawModules = (env.MODULES ?: params.APP)
            .split(',')
            .collect { it.trim() }
            .findAll { it }

          // --- If parent is packaging pom, never download jar for parent
          def parentPackaging = ''
          if (fileExists('pom.xml')) {
            def pom = new XmlSlurper(false, false).parseText(readFile('pom.xml'))
            parentPackaging = (pom.depthFirst().find { it.name() == 'packaging' }?.text() ?: '').trim()
          }

          def modules = rawModules
          if (parentPackaging == 'pom') {
            modules = rawModules.findAll { it != params.APP }
          }

          echo "DEBUG parentPackaging=${parentPackaging}"
          echo "DEBUG rawModules=${rawModules}"
          echo "DEBUG modulesForJar=${modules}"

          if (env.IS_MULTI == "true" && modules.isEmpty()) {
            error("IS_MULTI=true mais aucun module jar après exclusion du parent (MODULES=${env.MODULES})")
          }

          def hosts = env.TARGET_VMS.split(',')

          withCredentials([usernamePassword(credentialsId: env.ART_CRED, usernameVariable: 'ART_USER', passwordVariable: 'ART_PASS')]) {

            def resolveSnapshotFilename = { String repo, String groupPath, String artifactId, String version, String ext ->
              def metaUrl = "${env.EAI_ARTIFACTORY_URL}/${repo}/${groupPath}/${artifactId}/${version}/maven-metadata.xml"
              def xml = sh(
                returnStdout: true,
                script: """
                  set -euo pipefail
                  curl -fsS -u "${ART_USER}:${ART_PASS}" "${metaUrl}"
                """
              ).trim()

              def value = (xml =~ /<snapshotVersion>\\s*<extension>${ext}<\\/extension>\\s*(?:<classifier>[^<]+<\\/classifier>\\s*)?<value>([^<]+)<\\/value>/)
              if (value.find()) {
                def v = value.group(1)
                return "${artifactId}-${v}.${ext}"
              }

              def ts = (xml =~ /<timestamp>([^<]+)<\\/timestamp>/)
              def bn = (xml =~ /<buildNumber>([^<]+)<\\/buildNumber>/)
              if (ts.find() && bn.find()) {
                def baseVer = version.replaceAll(/-SNAPSHOT$/, "")
                return "${artifactId}-${baseVer}-${ts.group(1)}-${bn.group(1)}.${ext}"
              }
              return null
            }

            def download = { String url, String outFile ->
              sh """
                set -euo pipefail
                echo "Downloading: ${url}"
                curl -f -u "${ART_USER}:${ART_PASS}" -o "${outFile}" "${url}"
              """
            }

            // ✅ CONF: always global APP-config (even multi-module)
            def confArtifactId = "${params.APP}-config"

            // Download conf once per build (not per module)
            if (env.NEEDS_CONF == "true") {
              sh 'mkdir -p artifacts && rm -f artifacts/conf.zip || true'

              def confResolved = params.CONF_VERSION.endsWith("-SNAPSHOT")
                ? resolveSnapshotFilename(env.CONF_REPO, env.GROUP_PATH, confArtifactId, params.CONF_VERSION, "zip")
                : "${confArtifactId}-${params.CONF_VERSION}.zip"

              if (!confResolved) error("Impossible de résoudre le conf SNAPSHOT pour ${confArtifactId} ${params.CONF_VERSION}")

              def confUrl = "${env.EAI_ARTIFACTORY_URL}/${env.CONF_REPO}/${env.GROUP_PATH}/${confArtifactId}/${params.CONF_VERSION}/${confResolved}"
              download(confUrl, "artifacts/conf.zip")
            }

            modules.each { moduleArtifactId ->
              echo "===== DEPLOY MODULE: ${moduleArtifactId} ====="
              sh 'mkdir -p artifacts || true'

              if (env.NEEDS_JAR == "true") {
                def jarResolved = params.JAR_VERSION.endsWith("-SNAPSHOT")
                  ? resolveSnapshotFilename(env.JAR_REPO, env.GROUP_PATH, moduleArtifactId, params.JAR_VERSION, "jar")
                  : "${moduleArtifactId}-${params.JAR_VERSION}.jar"

                if (!jarResolved) error("Impossible de résoudre le jar SNAPSHOT pour ${moduleArtifactId} ${params.JAR_VERSION}")

                def jarUrl  = "${env.EAI_ARTIFACTORY_URL}/${env.JAR_REPO}/${env.GROUP_PATH}/${moduleArtifactId}/${params.JAR_VERSION}/${jarResolved}"
                def finalJar = "artifacts/${moduleArtifactId}-${params.JAR_VERSION}.jar"
                download(jarUrl, finalJar)
              }

              sh 'ls -lh artifacts || true'

              hosts.each { host ->
                sh """
                  set -euo pipefail
                  echo "=== Copy to ${host} (${moduleArtifactId}) ==="
                  scp ${env.SSH_OPTS} deploy_remote.sh ${env.EAI_SSH_USER}@"${host}:/tmp/deploy_remote.sh"
                  ${env.NEEDS_JAR == "true" ? "scp ${env.SSH_OPTS} artifacts/${moduleArtifactId}-${params.JAR_VERSION}.jar ${env.EAI_SSH_USER}@\"${host}:/tmp/app.jar\"" : "true"}
                  ${env.NEEDS_CONF == "true" ? "scp ${env.SSH_OPTS} artifacts/conf.zip ${env.EAI_SSH_USER}@\"${host}:/tmp/conf.zip\"" : "true"}

                  echo "=== Deploy on ${host} (${moduleArtifactId}) ==="
                  ssh ${env.SSH_OPTS} ${env.EAI_SSH_USER}@"${host}" "bash /tmp/deploy_remote.sh \
                    '${params.APP}' \
                    '${moduleArtifactId}' \
                    '${params.ENV}' \
                    '${params.INSTANCE}' \
                    '${params.DEPLOY_MODE}' \
                    '${params.JAR_VERSION ?: ''}' \
                    '${params.CONF_VERSION ?: ''}' \
                    '${env.IS_MULTI}'"
                """
              }
            }
          }
        }
      }
    }
  }
}
